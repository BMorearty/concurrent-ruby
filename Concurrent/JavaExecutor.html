def post(*args)
  raise ArgumentError.new('no block given') unless block_given?
  if running?
    @executor.submit{ yield(*args) }
    true
  else
    false
  end
rescue Java::JavaUtilConcurrent::RejectedExecutionException => ex
  raise RejectedExecutionError
end
