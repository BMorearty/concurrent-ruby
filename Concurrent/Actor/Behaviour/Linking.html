def on_envelope(envelope)
  case envelope.message
  when :link
    link envelope.sender
  when :unlink
    unlink envelope.sender
  else
    pass envelope
  end
end

def on_event(event)
  @linked.each { |a| a << event }
  @linked.clear if event == :terminated
  super event
end
